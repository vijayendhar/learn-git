pipeline {
   agent any
       stages {
          stage ("one") {
                    steps {
                       echo "This is stage 1 print"
                    }
           }
           stage ("Two") {
                     steps {
                         input("Do you want me to continue")
                     }
            }
           stage ("Three") 
	   {
                        when {
                              not{ 
                                    branch "master"
                              }
                         }

                       steps {

                               echo "This is stage 3 print"
                       }
             }
             stage ("Four")
	     {
                        if($Buildchoice == "Validate");
			
			steps
				{
                                bat mvn validate
                        	}
			else if($Buildchoice == "Compile");
			
			steps
			       {
                                  bat mvn compile
			        }
                        else if($Buildchoice == "Deploy");
			
			steps
				{
                                build VMothey_Test
				}
                        else
			steps
			        {
                              bat  mvn clean
				}
		}
           }
       }
